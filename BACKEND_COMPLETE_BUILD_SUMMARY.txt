# Register user

  -H "Content-Type: application/json" \
  -d '{
    "email": "user@example.com",
    "name": "Test User",
    "password": "test123",
    "confirmPassword": "test123"
  }'

  -H "Content-Type: application/json" \
  -d '{
    "email": "user@example.com",
    "password": "test123"
  }'

# Use token (replace TOKEN with actual token)
curl http://localhost:5000/api/auth/me \
  -H "Authorization: Bearer TOKEN"

# Search papers (public endpoint)
curl "http://localhost:5000/api/research/search?search=AI&limit=10"

# Create paper (requires auth)
curl -X POST http://localhost:5000/api/research \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer TOKEN" \
  -d '{
    "title": "AI in Healthcare",
    "authors": ["John Doe"],
    "type": "PAPER",
    "content": "Abstract...",
    "source": "https://arxiv.org/123"
  }'
